# 5.5 线性dp

## 5.5.1 什么是线性dp

线性dp主要用来描述以下的一类问题

> 当动态规划的状态转移方程满足以下格式
> f[x] = trans(f[x-1])
> 此时，我们称之为线性dp

## 5.5.2 代码模板

### 5.5.2.1 数字三角形

**朴素版**

```c++
#include <iostream>
using namespace std;

const int N = 505;
int s[N][N];
int f[N][N];

int main() {
    int n;
    cin >> n;
    for (int i = 0; i < n; i ++) {
        for (int j = 0; j <= i; j ++) {
            cin >> s[i][j];
        }
    }
    for (int i = n - 1; i >= 0; i --) {
        for (int j = 0; j <= i; j ++) {
            f[i][j] = max(f[i + 1][j], f[i + 1][j + 1]) + s[i][j];
        }
    }
    cout << f[0][0] << endl;
    return 0;
}
```

### 5.5.2.2 最长上升子序列

**朴素版**

原题：[最长上升子序列](https://www.acwing.com/problem/content/897/)

```c++
#include <iostream>
using namespace std;

const int N = 1e3 + 5;
int a[N];
int f[N]; // 以N为结尾的最长上升子序列的元素数

int main() {
    int n;
    cin >> n;
    for (int i = 1; i <= n; i ++) {
        cin >> a[i];
    }
    for (int i = 1; i <= n; i ++) {
        f[i] = 1;
        for (int j = 1; j < i; j ++) {
            if (a[i] > a[j])
                f[i] = max(f[i], f[j] + 1);
        }
    }
    int res = 1;
    for (int i = 1; i <= n; i ++) {
        res = max(res, f[i]);
    }
    cout << res << endl;
    return 0;
}
```

**栈优化版**

原题：[最长上升子序列Ⅱ](https://www.acwing.com/problem/content/898/)

```c++
#include <iostream>
#include <vector>
using namespace std;

const int N = 1e5 + 5;

vector<int> s(N), stk;

int binSearch(int t) {
    int l = 0, r = stk.size() - 1;
    while (l < r) {
        int mid = (l + r) / 2;
        if (stk[mid] >= t) r = mid;
        else l = mid + 1;
    }
    return l;
}

int main() {
    int n;
    cin >> n;
    for (int i = 0; i < n; i ++) cin >> s[i];

            stk.push_back(s[i]);

                stk[l] = s[i];
```

### 5.5.2.3 最长公共子序列

原题：[最长公共子序列](https://www.acwing.com/problem/content/899/)

```c++
#include <iostream>
using namespace std;

const int N = 1e3 + 5;
char a[N], b[N];
int f[N][N];

int main() {
    int n, m;
    cin >> n >> m;
    scanf("%s", a + 1);
    scanf("%s", b + 1);
    for (int i = 1; i <= n; i ++) {
        for (int j = 1; j <= m; j ++) {
            // a[i]不在公共子序列和b[j]不在公共子序列中的公共子序列长度的最大值
            f[i][j] = max(f[i - 1][j], f[i][j - 1]);
            // a[i]和b[j]都不在公共子序列的长度 + 1
            if (a[i] == b[j]) f[i][j] = max(f[i][j], f[i - 1][j - 1] + 1);
        }
    }
    cout << f[n][m] << endl;
    return 0;
}
```

### 5.5.2.4 最短编辑距离

原题：[最短编辑距离](https://www.acwing.com/problem/content/904/)

```c++
#include <iostream>
using namespace std;

const int N = 1e3 + 5;
char a[N], b[N];
int f[N][N];

int main() {
    int n, m;
    scanf("%d%s", &n, a + 1);
    scanf("%d%s", &m, b + 1);
    for (int i = 1; i <= n; i ++) f[i][0] = i; // 全部删除
    for (int i = 1; i <= m; i ++) f[0][i] = i; // 全部插入
    for (int i = 1; i <= n; i ++) {
        for (int j = 1; j <= m; j ++) {
            // 插入和删除中最小的
            f[i][j] = min(f[i - 1][j], f[i][j - 1]) + 1;
            f[i][j] = min(f[i][j], f[i - 1][j - 1] + (a[i] != b[j]));
        }
    }
    cout << f[n][m] << endl;
    return 0;
}
```
